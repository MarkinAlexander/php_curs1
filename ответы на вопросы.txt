<?
//объявляем переменные
$a=1;
$b=2;
//необходимо поменять местами что бы а=2 а b=1
$a=$a+$b; // a=2+1=3
$b=$a-$b; //b=3-1=2 то есть сейчас мы достигли нужного результата для b
$a=$a-$b; //a=3-2=1 так как в b сейчас другое значение, произведя операцию что и выше мы получим правильное решение
?>
